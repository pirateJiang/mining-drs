------------------------------------------------

"Characterize Threshold Crossing Event in Loop" Assign2
-------------------------------------------------------
drs_DurationUntilThresholdCrossing
	MN(drs_DurationUntilThresholdCrossing,
(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))<0)*(Eval(confExString_LowerLevelThreshold(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))-drs_Level(MN(Index,dim_NumberOfLevels)))/(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))+(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))==0))+(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))>0)*(Eval(confExString_UpperLevelThreshold(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))-drs_Level(MN(Index,dim_NumberOfLevels)))/(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))+(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))==0))+(Eval(confExString_LevelRate(MN(Index,dim_NumberOfLevels),drs_RateConfigurationNumber))==0||Index>dim_NumberOfLevels)*99999,
(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))<0)*(Eval(confExString_LowerTimerThreshold(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))-drs_Timer(MN(Index,dim_NumberOfTimers)))/(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))+(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))==0))+(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))>0)*(Eval(confExString_UpperTimerThreshold(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))-drs_Timer(MN(Index,dim_NumberOfTimers)))/(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))+(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))==0))+(Eval(confExString_TimerRate(MN(Index,dim_NumberOfTimers),drs_RateConfigurationNumber))==0||Index>dim_NumberOfTimers)*99999)

drs_ThresholdCrossingLevelOrTimerNumber
